1. Get VM kernel version
wget -c "https://cdn.kernel.org/pub/linux/kernel/v4.x/linux-4.9.36.tar.xz"

2. decompress kernel file
tar -Jxvf linux-4.9.36.tar.xz 

3. Backup origin kernel source

cp -a linux-4.9.36 linux-4.9.36.orig 

4. Setup VM default setting

cd linux-4.9.36
export ARCH=arm ;export CROSS_COMPILE=/usr/bin/arm-linux-gnueabihf-
make vexpress_defconfig 
make menuconfig
